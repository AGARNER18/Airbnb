--**********************************************Drop Tables***********************************
--in specific order so that child tables are dropped before parent tables
DROP TABLE REVIEWS;
DROP TABLE LISTINGS;
DROP TABLE HOST;
DROP TABLE LOCATION;


--******************************************Create Tables************************************
--create location table
CREATE HADOOP TABLE IF NOT EXISTS LOCATION 
    (neighborhood_id INT NOT NULL,
    neighborhood VARCHAR(18),
    state VARCHAR(2),
    median_income INT,
    median_home INT,
    population INT,
    median_age NUMERIC(4,2),
    crime_index INT,
    CONSTRAINT pk_neighbhorhood_id PRIMARY KEY (neighborhood_id)
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

--create host table
CREATE HADOOP TABLE IF NOT EXISTS HOST 
   (host_id INT NOT NULL,
    host_name VARCHAR(40),
    host_location VARCHAR(46),
    host_response_time VARCHAR(18),
    host_response_rate NUMERIC(3, 2),
    superhost VARCHAR(1),
    host_listing_count INT,
    cancellation_policy VARCHAR(8),
    years_host INT,
    CONSTRAINT pk_host_id PRIMARY KEY (host_id),
    CONSTRAINT check_superhost check (superhost in ('t', 'f'))
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

CREATE HADOOP TABLE IF NOT EXISTS LISTINGS (
    listing_id INT NOT NULL,
    host_id INT,
    neighborhood_id INT,
    latitude NUMERIC(10, 8),
    longitude NUMERIC(11, 8),
    property_type VARCHAR(15),
    room_type VARCHAR(15),
    accommodates INT,
    price INT,
    min_nights INT,
    max_nights INT,
    availability INT,
    instant_book VARCHAR(1),
    CONSTRAINT pk_listing_id PRIMARY KEY (listing_id),
    CONSTRAINT fk_listing_id FOREIGN KEY (host_id) references HOST(host_id),
    CONSTRAINT fk_neighborhood_id FOREIGN KEY (neighborhood_id) references LOCATION(neighborhood_id),
    CONSTRAINT check_instant_book check (instant_book in ('t', 'f')),
    CONSTRAINT check_room_type check (room_type in ('Entire home/apt', 'Private room', 'Shared room'))
)ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

CREATE HADOOP TABLE IF NOT EXISTS REVIEWS (
    review_id INT NOT NULL,
    listing_id INT,
    number_of_reviews INT,
    overall_rating INT,
    listing_accuracy INT,
    cleanliness INT,
    check_in INT,
    communication INT,
    location INT,
    value INT, 
    CONSTRAINT pk_review_id PRIMARY KEY (review_id),
    CONSTRAINT fk_listing_id FOREIGN KEY (listing_id) references LISTINGS(listing_id)
)ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

SELECT L.PRICE, L.ACCOMMODATES, L.ROOM_TYPE, L.PROPERTY_TYPE, N.NEIGHBORHOOD, N.STATE, R.OVERALL_RATING

SELECT NEIGHBORHOOD, ACCOMMODATES, MIN(PRICE) LOWEST, MAX(PRICE) HIGHEST, AVG(PRICE) AS "AVERAGE PRICE", COUNT(*)
FROM BY_AREA
WHERE STATE='CA' AND ACCOMMODATES < 6 AND PROPERTY_TYPE= 'House' AND ROOM_TYPE='Entire home/apt'
GROUP BY NEIGHBORHOOD, ACCOMMODATES
HAVING COUNT(*) > 1
ORDER BY ACCOMMODATES DESC;

--see min, max and average price of each neighborhood in LA
